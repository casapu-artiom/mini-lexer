PROGRAM -> func main ( ) { STATEMENTS }
STATEMENTS -> STATEMENT | STATEMENTS STATEMENT
STATEMENTS ->
STATEMENT -> VARIABLE_DECL | ASSIGNMENT | READ | WRITE | CONDITIONAL_STATEMENT
VARIABLE_DECL -> TYPE_IDENTIFIER id ;
TYPE_IDENTIFIER -> integer | char
ASSIGNMENT -> id = ARITHMETIC_EXPRESSION ; | id = constant ; | id = id ;
READ -> read ( id ) ;
WRITE -> write ( id ) ;
OPERAND -> id | constant
CONDITIONAL_STATEMENT -> if ( BOOLEAN_EXPRESSION ) { STATEMENTS }
BOOLEAN_EXPRESSION -> OPERAND BOOLEAN_OPERATOR OPERAND
BOOLEAN_OPERATOR -> == | != | > | < | >= | <=
ARITHMETIC_EXPRESSION -> ARITHMETIC_EXPRESSION + TERMEN | TERMEN
TERMEN -> TERMEN * FACTOR | FACTOR
FACTOR -> OPERAND | ( ARITHMETIC_EXPRESSION )
